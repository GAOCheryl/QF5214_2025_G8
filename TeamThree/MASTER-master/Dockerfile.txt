FROM python:3.9-slim

WORKDIR /app

# Copy requirements and install dependencies if needed
COPY requirements.txt .
RUN pip install -r requirements.txt

# Copy your Python script
COPY your_script.py .

# Install cron
RUN apt-get update && apt-get -y install cron

# Create crontab file
RUN echo "0 18 * * * python /app/your_script.py >> /var/log/cron.log 2>&1" > /etc/cron.d/script-cron
RUN chmod 0644 /etc/cron.d/script-cron

# Apply cron job
RUN crontab /etc/cron.d/script-cron

# Create log file
RUN touch /var/log/cron.log

# Run cron in foreground
CMD cron && tail -f /var/log/cron.log
